// עדכון פונקציות ה-JavaScript בתוך admin.html async function loadAdminPosts() { const response = await fetch('/api/posts'); const posts = await response.json(); const tbody = document.getElementById('adminTableBody'); tbody.innerHTML = posts.map((post) => ` ${post.date} ${post.author} ${post.content} `).join(''); } async function deletePost(id) { if(confirm("למחוק הודעה זו?")) { await fetch(`/api/posts/${id}`, { method: 'DELETE' }); loadAdminPosts(); } }
